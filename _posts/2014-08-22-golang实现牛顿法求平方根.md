    package main
    
    import (
    	"fmt"
    )
    
    func Sqrt(x float64) float64 {
    	var z float64 = 1.5
    	for i := 0; i < 3; i++ {
    		z = z - (z*z-x)/2.0/z
    		fmt.Println(z)
    	}
    	return z
    }
    
    func main() {
    	fmt.Println(Sqrt(2))
    }
